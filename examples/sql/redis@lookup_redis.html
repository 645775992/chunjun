INSERT INTO sink
SELECT
    u.id
     , u.name
     , u.money
     , u.dateone
     , u.age
     , u.datethree
     , u.datesix
     , u.datenigth
     , u.dtdate
     , u.dttime


     , s.id        as aid
     , s.name      as aname
     , s.money     as amoney
     , s.dateone   as adateone
     , s.age       as aage
     , s.datethree as adatethree
     , s.datesix   as adatesix
     , s.datenigth as adatenigth
     , s.dtdate    as adtdate
     , s.dttime    as adttime


     , s.afloat
     , s.adouble
     , s.aboolean
     , s.abigint
     , s.atinyint
     , s.avarchar
     , s.asmallint
from source u
         left join side FOR SYSTEM_TIME AS OF u.PROCTIME AS s
                   on u.id = s.id;
